𑍐
𑌅𑍨𑌗𑍍𑌨𑍩 𑌆𑍧 𑌯𑌾𑍨𑌹𑌿 𑌵𑍀𑍩𑌤𑍧𑌯𑍇𑍨 𑌗𑍃𑌣𑌾𑍩𑌨𑍋𑍨 𑌹𑍩𑌵𑍍𑌯𑍧𑌦𑌾𑍨𑌤𑌯𑍇 । 𑌨𑌿𑍧 𑌹𑍋𑌤𑌾𑍨𑍦 𑌸𑌤𑍍𑌸𑌿 𑌬𑍩𑌰𑍍𑌹𑌿𑍧𑌷𑌿𑍨 ॥ ௧ ॥
𑌤𑍍𑌵𑍧𑌮𑍨𑌗𑍍𑌨𑍇 𑌯𑍩𑌜𑍍𑌞𑌾𑍨𑌨𑌾𑍩𑍞 𑌹𑍋𑍨𑌤𑌾𑍩 𑌵𑌿𑍧 𑌶𑍍𑌵𑍇𑍨𑌷𑌾𑍞 𑌹𑌿𑍩𑌤𑌃𑍨 । 𑌦𑍇𑍩𑌵𑍇𑍨𑌭𑌿𑍩𑌰𑍍𑌮𑌾𑍧𑌨𑍁𑍨𑌷𑍇𑍩 𑌜𑍧𑌨𑍇𑍨 ॥ ௨ ॥

𑌅𑌰𑍍𑌚𑌿𑌕𑌾
𑌅𑍧𑌸𑌾𑍨𑌵𑌿𑌦𑍇𑍩𑌵𑍧𑌂𑌗𑍋𑌋𑍨𑍦𑌜𑍀𑌕𑍩𑌮𑍨𑌂𑌧𑍋𑍩𑍱𑌨𑍍𑌯𑍨𑍦𑌸𑍍𑌮𑌿𑍩𑍧𑌨𑍍𑌨𑌿𑌂𑌦𑍍𑌰𑍋𑍨𑌜𑍩𑌨𑍁𑍧𑌷𑍇𑍨𑌮𑍁𑌵𑍋𑌚 ।
𑌬𑍋𑍧𑌧𑌾𑍨𑌮𑌸𑌿𑌤𑍍𑌵𑌾𑌹𑌰𑍍𑌯𑌶𑍍𑌵𑌯𑍩𑌜𑍍𑌞𑍈𑍧𑌰𑍍𑌬𑍋𑌧𑌾𑍨𑍦𑌨𑍩𑌃𑌸𑍍𑌤𑍋𑍨𑌮𑍩𑌮𑍧𑌂𑌧𑍨𑌸𑍋𑍨𑌮𑍧𑌦𑍇𑍨𑌷𑍁 ॥ ௧ ॥

𑌗𑌣𑌃
𑌪𑍍𑌰𑍇𑍪𑍦𑌷𑍍𑌠𑍫𑌂𑌵𑌃‐𑌓𑍪𑌹𑌾𑌇 ॥
𑌅𑍧𑌤𑌾𑌽௨௩𑌇𑌥𑍀𑍨𑌮𑍍‐𑌸𑍍𑌤𑍁𑌷𑌾𑌇𑌮𑌿𑌤𑍍𑌰𑌾𑌽௩𑌮𑍍‐𑌇𑍧𑌵𑌾𑌽᳒௨𑌪𑍍𑌰𑌾𑍩𑌽௨௩௪𑌯𑌾𑍫𑌮𑍍 ॥
𑌔𑍨𑍦𑌹𑍋𑌽௧𑌇‐𑌅𑌗𑍍𑌨𑍇𑍨𑍦𑌰𑌾𑍧𑌥𑌾𑌽௨௩𑌮𑍍 ॥
𑌨𑌾𑍧𑌽௨௩𑌵𑍇𑍪𑌽௩‐𑌦𑌾𑍨𑌽௩௪௫𑌯𑍋𑌽௧᳓𑌹𑌾𑍫𑌇 ॥
